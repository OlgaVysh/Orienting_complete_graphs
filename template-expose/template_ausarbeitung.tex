\begin{algorithm}
	\caption{Breitensuche (BFS)}
	\label{alg:BFS}
	\begin{algorithmic} []
		\REQUIRE Graph $G$, Startknoten $s$, leere Warteschlange $Q$
		\ENSURE BFS
		
		\STATE	$Q$.enqueue($s$)			
		\STATE mark $s$ as visited
		\WHILE{$Q$ is not empty}
		\STATE	$v$ = $Q$.dequeue( )			
		\FORALL{neighbors $w$ of $v$}
		\IF{$w$ is not visited}
		\STATE	$Q$.enqueue($w$)
		\STATE	mark $w$ as visited
		\ENDIF
		\ENDFOR
		\ENDWHILE
	\end{algorithmic}
\end{algorithm}
